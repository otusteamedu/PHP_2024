SELECT
    i.relname AS имя_индекса,
    s.idx_scan
FROM
    pg_stat_user_indexes s
        JOIN
    pg_index x ON s.indexrelid = x.indexrelid
    JOIN
    pg_class i ON s.indexrelid = i.oid
order by idx_scan  asc
limit 5;
